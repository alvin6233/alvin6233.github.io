oracle递归查询

2018-12-18 积累数据存储 阅读量

工作中遇到了递归查询部门树的需求，之前虽然用过，但隔的时间久了记不清具体用法。在这里mark一下，重新熟悉下

查看表结构

执行 desc dept_tree;

插入数据

select [column, …] from tablename
start with [condition1]
connect by [condition2]
where [condition3];

select id, name,sys_connect_by_path(name, '=>') namejoins, level
from dept_tree
start with dept_tree.id=17
connect by  id = prior parent


select id, name,sys_connect_by_path(name, '=>') namejoins, level
from dept_tree
start with dept_tree.id=17
connect by  prior id = parent


select id, name,sys_connect_by_path(name, '=>') namejoins, level
from dept_tree
start with 1=1
connect by  prior id = parent

select id, name,sys_connect_by_path(name, ‘=>’) namejoins, level
from dept_tree
start with 1=1
connect by id = prior parent
```
